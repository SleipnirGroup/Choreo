114 DifferentialSample Interpolate(const DifferentialSample& endValue,
115 units::second_t t) const {
116 units::scalar_t scale = (t - timestamp) / (endValue.timestamp - timestamp);
117
118 // Integrate the acceleration to get the rest of the state, since linearly
119 // interpolating the state gives an inaccurate result if the accelerations
120 // are changing between states
121 Eigen::Vector<double, 6> initialState{x.value(), y.value(),
122 heading.value(), vl.value(),
123 vr.value(), omega.value()};
124
125 // FIXME: this means the function cant be constexpr without c++23
126 std::function<Eigen::Vector<double, 6>(Eigen::Vector<double, 6>,
127 Eigen::Vector<double, 3>)>
128 f = [](Eigen::Vector<double, 6> state, Eigen::Vector<double, 3> input) {
129 // state = [x, y, θ, vₗ, vᵣ, ω]
130 // input = [aₗ, aᵣ, α]
131 //
132 // v = (vₗ + vᵣ)/2
133 //
134 // ẋ = v cosθ
135 // ẏ = v sinθ
136 // θ̇ = ω
137 // v̇ₗ = aₗ
138 // v̇ᵣ = aᵣ
139 // ω̇ = α
140 auto θ = state(2, 0);
141 auto vl = state(3, 0);
142 auto vr = state(4, 0);
143 auto ω = state(5, 0);
144 auto al = input(0, 0);
145 auto ar = input(1, 0);
146 auto α = input(2, 0);
147 auto v = (vl + vr) / 2;
148 return Eigen::Vector<double, 6>{
149 v * std::cos(θ), v * std::sin(θ), ω, al, ar, α};
150 };
151
152 units::second_t τ = t - timestamp;
153 auto sample = frc::RKDP(
154 f, initialState,
155 Eigen::Vector<double, 3>(al.value(), ar.value(), alpha.value()), τ);
156
157 return DifferentialSample{
158 wpi::Lerp(timestamp, endValue.timestamp, scale),
159 units::meter_t{sample(0, 0)},
160 units::meter_t{sample(1, 0)},
161 units::radian_t{sample(2, 0)},
162 units::meters_per_second_t{sample(3, 0)},
163 units::meters_per_second_t{sample(4, 0)},
164 units::radians_per_second_t{sample(5, 0)},
165 al,
166 ar,
167 alpha,
168 wpi::Lerp(fl, endValue.fl, scale),
169 wpi::Lerp(fr, endValue.fr, scale),
170 };
171 }

228 constexpr bool operator==(const DifferentialSample& other) const {
229 constexpr double epsilon = 1e-6;
230
231 auto compare_units = [epsilon](const auto& a, const auto& b) {
232 using UnitType =
233 std::remove_const_t<std::remove_reference_t<decltype(a)>>;
234 return units::math::abs(a - b) < UnitType(epsilon);
235 };
236
237 return compare_units(timestamp, other.timestamp) &&
238 compare_units(x, other.x) && compare_units(y, other.y) &&
239 compare_units(heading, other.heading) &&
240 compare_units(vl, other.vl) && compare_units(vr, other.vr) &&
241 compare_units(omega, other.omega) && compare_units(al, other.al) &&
242 compare_units(ar, other.ar) && compare_units(alpha, other.alpha) &&
243 compare_units(fl, other.fl) && compare_units(fr, other.fr);
244 }
