95template <int Year = kDefaultYear>
96constexpr auto GetFlipperForYear() {
97 constexpr bool yearInMap = [] {
98 try {
99 [[maybe_unused]]
100 auto checked = flipperMap.at(Year);
101 return true;
102 } catch (...) {
103 return false;
104 }
105 }();
106
107 if constexpr (!yearInMap) {
108 static_assert(yearInMap, "Year not found in flipperMap");
109 } else {
110 constexpr auto flipperType = flipperMap.at(Year);
111 if constexpr (flipperType == FlipperType::RotateAround) {
112 return RotateAroundFlipper{};
113 } else if constexpr (flipperType == FlipperType::Mirrored) {
114 return MirroredFlipper{};
115 } else {
116 static_assert(flipperType == FlipperType::RotateAround ||
117 flipperType == FlipperType::Mirrored,
118 "Invalid FlipperType in flipperMap");
119 }
120 }
121}
