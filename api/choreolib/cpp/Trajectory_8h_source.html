232 std::optional<Trajectory<SampleType>> GetSplit(int splitIndex) const {
233 // Assumption: splits[splitIndex] is a valid index of samples.
234 if (splitIndex < 0 || splitIndex >= splits.size()) {
235 return std::nullopt;
236 }
237
238 int start = splits[splitIndex];
239 int end = (splitIndex + 1 < splits.size()) ? splits[splitIndex + 1] + 1
240 : samples.size();
241
242 auto sublist =
243 std::vector<SampleType>(samples.begin() + start, samples.begin() + end);
244 // Empty section should not be achievable (would mean malformed splits
245 // array), but is handled for safety
246 if (sublist.size() == 0) {
247 return Trajectory<SampleType>{
248 name + "[" + std::to_string(splitIndex) + "]", {}, {}, {}};
249 }
250 // Now we know sublist.size() >= 1
251 units::second_t startTime = sublist.front().GetTimestamp();
252 units::second_t endTime = sublist.back().GetTimestamp();
253
254 auto offsetSamples =
255 sublist | std::views::transform([startTime](const SampleType& s) {
256 return s.OffsetBy(-startTime);
257 });
258
259 auto filteredEvents =
260 events | std::views::filter([startTime, endTime](const auto& e) {
261 return e.timestamp >= startTime && e.timestamp <= endTime;
262 }) |
263 std::views::transform(
264 [startTime](const auto& e) { return e.OffsetBy(-startTime); });
265
266 return Trajectory<SampleType>{
267 name + "[" + std::to_string(splitIndex) + "]",
268 std::vector<SampleType>(offsetSamples.begin(), offsetSamples.end()),
269 {},
270 std::vector<EventMarker>(filteredEvents.begin(), filteredEvents.end())};
271 }

277 bool operator==(const Trajectory<SampleType>& other) const {

298 }

300 return true;
301 }

316 std::optional<SampleType> SampleInternal(units::second_t timestamp) const {
317 if (samples.size() == 0) {
318 return {};
319 }
320 if (samples.size() == 1) {
321 return samples[0];
322 }
323 if (timestamp < samples[0].GetTimestamp()) {
324 return GetInitialSample();
325 }
326 if (timestamp >= GetTotalTime()) {
327 return GetFinalSample();
328 }
329
330 int low = 0;
331 int high = samples.size() - 1;
332
333 while (low != high) {
334 int mid = (low + high) / 2;
335 if (samples[mid].GetTimestamp() < timestamp) {
336 low = mid + 1;
337 } else {
338 high = mid;
339 }
340 }
341
342 if (low == 0) {
343 return samples[low];
344 }
345
346 SampleType behindState = samples[low - 1];
347 SampleType aheadState = samples[low];
348
349 if ((aheadState.GetTimestamp() - behindState.GetTimestamp()) < 1e-6_s) {
350 return aheadState;
351 }
352
353 return behindState.Interpolate(aheadState, timestamp);
354 }
