46 explicit PointLineRegionConstraint(Translation2d robot_point,
47 Translation2d field_line_start,
48 Translation2d field_line_end, Side side)
49 : m_robot_point{std::move(robot_point)},

64 void apply(
65 slp::Problem<double>& problem, const Pose2v<double>& pose,
66 [[maybe_unused]] const Translation2v<double>& linear_velocity,
67 [[maybe_unused]] const slp::Variable<double>& angular_velocity,
68 [[maybe_unused]] const Translation2v<double>& linear_acceleration,
69 [[maybe_unused]] const slp::Variable<double>& angular_acceleration) {
70 auto point = pose.translation() + m_robot_point.rotate_by(pose.rotation());
71
72 // Determine which side of the start-end field line a point is on.
73 //
74 // The cross product a x b = |a|₂|b|₂sinθ for a and b vectors with the same
75 // tail. If a x b > 0, b is to the left of a.
76 //
77 // b
78 // ^
79 // |
80 // -----> a
81 //
82 //
83 // If a x b < 0, b is to the right of a.
84 //
85 // -----> a
86 // |
87 // v
88 // b
89 //
90 // Let a be the field line start -> end and let b be the point start ->
91 // point.
92 //
93 // cross > 0 means point is left of line (above)
94 // cross = 0 means point is on line
95 // cross < 0 means point is right of line (below)
96 auto line = m_field_line_end - m_field_line_start;
97 auto start_to_point = point - m_field_line_start;
98 auto cross = line.cross(start_to_point);
99
100 switch (m_side) {
101 case Side::ABOVE:
102 problem.subject_to(cross > 0);
103 break;
104 case Side::BELOW:
105 problem.subject_to(cross < 0);
106 break;
107 case Side::ON:
108 problem.subject_to(cross == 0);
109 break;
110 }
111 }
112
113 private:
114 Translation2d m_robot_point;
115 Translation2d m_field_line_start;
116 Translation2d m_field_line_end;
117 Side m_side;
118};
