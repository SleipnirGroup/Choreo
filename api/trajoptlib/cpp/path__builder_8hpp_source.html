181 void sgmt_constraint(size_t from_index, size_t to_index,
182 const Constraint& constraint) {
183 assert(from_index < to_index);
184
185 new_wpts(to_index);
186 path.waypoints.at(from_index).waypoint_constraints.push_back(constraint);
187 for (size_t index = from_index + 1; index <= to_index; ++index) {
188 path.waypoints.at(index).waypoint_constraints.push_back(constraint);
189 path.waypoints.at(index).segment_constraints.push_back(constraint);
190 }

256 void new_wpts(size_t final_index) {
257 if (final_index < path.waypoints.size()) {
258 return;
259 }
260
261 for (size_t i = path.waypoints.size(); i <= final_index; ++i) {
262 path.waypoints.emplace_back();
263 initial_guess_points.emplace_back(std::vector<Pose2d>{{0.0, 0.0, {0.0}}});
264 if (i != 0) {
265 control_interval_counts.push_back(40);
266 }
267 }
268 }
