102 void sgmt_initial_guess_points(size_t from_index,

121 void pose_wpt(size_t index, double x, double y, double heading) {
122 wpt_constraint(index, PoseEqualityConstraint{x, y, heading});
123 wpt_initial_guess_point(index, {x, y, {heading}});

135 void translation_wpt(size_t index, double x, double y,

145 void wpt_constraint(size_t index, const Constraint& constraint) {
146 new_wpts(index);

148 }
149
155 void sgmt_constraint(size_t from_index, size_t to_index,
156 const Constraint& constraint) {
157 assert(from_index < to_index);
158
159 new_wpts(to_index);
160 path.waypoints.at(from_index).waypoint_constraints.push_back(constraint);
161 for (size_t index = from_index + 1; index <= to_index; ++index) {
162 path.waypoints.at(index).waypoint_constraints.push_back(constraint);
163 path.waypoints.at(index).segment_constraints.push_back(constraint);
164 }
165 }

174 void add_callback(

183 Path<Drivetrain, Solution>& get_path() { return path; }

189 Solution calculate_linear_initial_guess() const {

219 void new_wpts(size_t final_index) {
220 if (final_index < path.waypoints.size()) {
221 return;
222 }
223
224 for (size_t i = path.waypoints.size(); i <= final_index; ++i) {
225 path.waypoints.emplace_back();
226 initial_guess_points.emplace_back(std::vector<Pose2d>{{0.0, 0.0, {0.0}}});
227 if (i != 0) {
228 control_interval_counts.push_back(40);
229 }
230 }
231 }
