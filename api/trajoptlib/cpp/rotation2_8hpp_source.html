166template <typename LHS, typename RHS>
167 requires slp::SleipnirType<LHS> && (!slp::SleipnirType<RHS>)
168auto operator==(const Rotation2<LHS>& lhs, const Rotation2<RHS>& rhs) {
169 using Scalar = typename std::decay_t<LHS>::Scalar;
170
171 std::vector<slp::EqualityConstraints<Scalar>> constraints;
172
173 // Constrain angle equality on manifold.
174 //
175 // Let lhs = <cos(a), sin(a)>.
176 // Let rhs = <cos(b), sin(b)>.
177 //
178 // If the angles are equal, the angle between the unit vectors should be
179 // zero.
180 //
181 // lhs x rhs = ‖lhs‖ ‖rhs‖ sin(angleBetween)
182 // = 1 * 1 * 0
183 // = 0
184 //
185 // NOTE: angleBetween = π rad would be another solution
186 constraints.emplace_back(lhs.cos() * rhs.sin() - lhs.sin() * rhs.cos() ==
187 Scalar(0));
188
189 // Require that lhs and rhs are unit vectors if they contain autodiff
190 // variables, since their values can change.
191 constraints.emplace_back(lhs.cos() * lhs.cos() + lhs.sin() * lhs.sin() ==
192 Scalar(1));
193
194 return slp::EqualityConstraints{constraints};
195}
