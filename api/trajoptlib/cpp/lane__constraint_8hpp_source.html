26 LaneConstraint(Translation2d center_line_start, Translation2d center_line_end,
27 double tolerance)
28 : m_top_line{[&] {
29 if (tolerance != 0.0) {
30 double dx = center_line_end.x() - center_line_start.x();
31 double dy = center_line_end.y() - center_line_start.y();
32 double dist = std::hypot(dx, dy);
33 auto offset = Translation2d{0.0, tolerance}.rotate_by(
34 Rotation2d{dx / dist, dy / dist});
35
36 return PointLineRegionConstraint{{0.0, 0.0},
37 center_line_start + offset,
38 center_line_end + offset,
39 Side::BELOW};
40 } else {
41 return PointLineRegionConstraint{
42 {0.0, 0.0}, center_line_start, center_line_end, Side::ON};
43 }
44 }()},
45 m_bottom_line{[&]() -> std::optional<PointLineRegionConstraint> {
46 if (tolerance != 0.0) {
47 double dx = center_line_end.x() - center_line_start.x();
48 double dy = center_line_end.y() - center_line_start.y();
49 double dist = std::hypot(dx, dy);
50 auto offset = Translation2d{0.0, tolerance}.rotate_by(
51 Rotation2d{dx / dist, dy / dist});
52
53 return PointLineRegionConstraint{{0.0, 0.0},
54 center_line_start - offset,
55 center_line_end - offset,
56 Side::ABOVE};
57 } else {
58 return std::nullopt;
59 }
60 }()} {}
