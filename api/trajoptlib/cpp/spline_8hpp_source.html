44 std::optional<PoseWithCurvature> get_point(double t) const {
45 Eigen::Vector<double, Degree + 1> polynomial_bases;
46
47 // Populate the polynomial bases
48 for (int i = 0; i <= Degree; i++) {
49 polynomial_bases(i) = std::pow(t, Degree - i);
50 }
51
52 // This simply multiplies by the coefficients. We need to divide out t some
53 // n number of times where n is the derivative we want to take.
54 Eigen::Vector<double, 6> combined = coefficients() * polynomial_bases;
55
56 double dx, dy, ddx, ddy;
57
58 // If t = 0, all other terms in the equation cancel out to zero. We can use
59 // the last x‚Å∞ term in the equation.
60 if (t == 0.0) {
61 dx = coefficients()(2, Degree - 1);
62 dy = coefficients()(3, Degree - 1);
63 ddx = coefficients()(4, Degree - 2);
64 ddy = coefficients()(5, Degree - 2);
65 } else {
66 // Divide out t for first derivative.
67 dx = combined[2] / t;
68 dy = combined[3] / t;
69
70 // Divide out t for second derivative.
71 ddx = combined[4] / t / t;
72 ddy = combined[5] / t / t;
73 }
74
75 if (std::hypot(dx, dy) < 1e-6) {
76 return std::nullopt;
77 }
78
79 // Find the curvature.
80 const auto curvature =
81 (dx * ddy - ddx * dy) / ((dx * dx + dy * dy) * std::hypot(dx, dy));
82
83 return PoseWithCurvature{
84 {Translation2d{combined[0], combined[1]}, Rotation2d{dx, dy}},
85 curvature};
86 }
