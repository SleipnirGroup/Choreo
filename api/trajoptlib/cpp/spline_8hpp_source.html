50 std::optional<PoseWithCurvature> get_point(double t) const {
51 Eigen::Vector<double, Degree + 1> polynomial_bases;
52
53 // Populate the polynomial bases
54 for (int i = 0; i <= Degree; i++) {
55 polynomial_bases(i) = std::pow(t, Degree - i);
56 }
57
58 // This simply multiplies by the coefficients. We need to divide out t some
59 // n number of times where n is the derivative we want to take.
60 Eigen::Vector<double, 6> combined = coefficients() * polynomial_bases;
61
62 double dx, dy, ddx, ddy;
63
64 // If t = 0, all other terms in the equation cancel out to zero. We can use
65 // the last x⁰ term in the equation.
66 if (t == 0.0) {
67 dx = coefficients()(2, Degree - 1);
68 dy = coefficients()(3, Degree - 1);
69 ddx = coefficients()(4, Degree - 2);
70 ddy = coefficients()(5, Degree - 2);
71 } else {
72 // Divide out t for first derivative.
73 dx = combined[2] / t;
74 dy = combined[3] / t;
75
76 // Divide out t for second derivative.
77 ddx = combined[4] / t / t;
78 ddy = combined[5] / t / t;
79 }
80
81 if (std::hypot(dx, dy) < 1e-6) {
82 return std::nullopt;
83 }
84
85 // Find the curvature.
86 const auto curvature =
87 (dx * ddy - ddx * dy) / ((dx * dx + dy * dy) * std::hypot(dx, dy));
88
89 return PoseWithCurvature{
90 {Translation2d{combined[0], combined[1]}, Rotation2d{dx, dy}},
91 curvature};
92 }
