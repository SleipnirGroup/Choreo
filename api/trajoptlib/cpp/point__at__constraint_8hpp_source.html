51 void apply(
52 slp::Problem<double>& problem, const Pose2v<double>& pose,
53 [[maybe_unused]] const Translation2v<double>& linear_velocity,
54 [[maybe_unused]] const slp::Variable<double>& angular_velocity,
55 [[maybe_unused]] const Translation2v<double>& linear_acceleration,
56 [[maybe_unused]] const slp::Variable<double>& angular_acceleration) {
57 // dx,dy = desired heading
58 // ux,uy = unit vector of desired heading
59 // hx,hy = heading
60 // dot = dot product of ux,uy and hx,hy
61 //
62 // constrain dot to cos(1.0), which is colinear
63 // and cos(theta_tolerance)
64 auto dx = m_field_point.x() - pose.x();
65 auto dy = m_field_point.y() - pose.y();
66 auto dot = pose.rotation().cos() * dx + pose.rotation().sin() * dy;
67 if (!m_flip) {
68 // dot close to 1 * hypot (point toward)
69 problem.subject_to(dot >=
70 std::cos(m_heading_tolerance) * slp::hypot(dx, dy));
71 } else {
72 // dot close to -1 * hypot (point away)
73 problem.subject_to(dot <=
74 -std::cos(m_heading_tolerance) * slp::hypot(dx, dy));
75 }
76 }
77
78 private:
79 Translation2d m_field_point;
80 double m_heading_tolerance;
81 bool m_flip;
82};
