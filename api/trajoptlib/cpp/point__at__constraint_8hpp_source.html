45 void apply(
46 slp::Problem<double>& problem, const Pose2v<double>& pose,
47 [[maybe_unused]] const Translation2v<double>& linear_velocity,
48 [[maybe_unused]] const slp::Variable<double>& angular_velocity,
49 [[maybe_unused]] const Translation2v<double>& linear_acceleration,
50 [[maybe_unused]] const slp::Variable<double>& angular_acceleration) {
51 // dx,dy = desired heading
52 // ux,uy = unit vector of desired heading
53 // hx,hy = heading
54 // dot = dot product of ux,uy and hx,hy
55 //
56 // constrain dot to cos(1.0), which is colinear
57 // and cos(theta_tolerance)
58 auto dx = m_field_point.x() - pose.x();
59 auto dy = m_field_point.y() - pose.y();
60 auto dot = pose.rotation().cos() * dx + pose.rotation().sin() * dy;
61 if (!m_flip) {
62 // dot close to 1 * hypot (point toward)
63 problem.subject_to(dot >=
64 std::cos(m_heading_tolerance) * slp::hypot(dx, dy));
65 } else {
66 // dot close to -1 * hypot (point away)
67 problem.subject_to(dot <=
68 -std::cos(m_heading_tolerance) * slp::hypot(dx, dy));
69 }
70 }
71
72 private:
73 Translation2d m_field_point;
74 double m_heading_tolerance;
75 bool m_flip;
76};
