27 static std::array<Spline<3>::ControlVector, 2>
28 cubic_control_vectors_from_waypoints(
29 const Pose2d& start, const std::vector<Translation2d>& interior_waypoints,
30 const Pose2d& end) {
31 double scalar;
32 if (interior_waypoints.empty()) {
33 scalar = 1.2 * start.translation().distance(end.translation());
34 } else {
35 scalar = 1.2 * start.translation().distance(interior_waypoints.front());
36 }
37 const auto initial_control_vector = cubic_control_vector(scalar, start);
38 if (!interior_waypoints.empty()) {
39 scalar = 1.2 * end.translation().distance(interior_waypoints.back());
40 }
41 const auto final_control_vector = cubic_control_vector(scalar, end);
42 return {initial_control_vector, final_control_vector};
43 }

59 static std::vector<CubicHermiteSpline> cubic_splines_from_control_vectors(
60 const Spline<3>::ControlVector& start,
61 std::vector<Translation2d> waypoints,
62 const Spline<3>::ControlVector& end) {
63 std::vector<CubicHermiteSpline> splines;
64
65 std::array<double, 2> x_initial = start.x;
66 std::array<double, 2> y_initial = start.y;
67 std::array<double, 2> x_final = end.x;
68 std::array<double, 2> y_final = end.y;
69
70 if (waypoints.size() > 1) {
71 waypoints.emplace(waypoints.begin(),
72 Translation2d{x_initial[0], y_initial[0]});
73 waypoints.emplace_back(Translation2d{x_final[0], y_final[0]});
74
75 // Populate tridiagonal system for clamped cubic
76 /* See:
77 https://www.uio.no/studier/emner/matnat/ifi/nedlagte-emner/INF-MAT4350/h08
78 /undervisningsmateriale/chap7alecture.pdf
79 */
80
81 // Above-diagonal of tridiagonal matrix, zero-padded
82 std::vector<double> a;
83 // Diagonal of tridiagonal matrix
84 std::vector<double> b(waypoints.size() - 2, 4.0);
85 // Below-diagonal of tridiagonal matrix, zero-padded
86 std::vector<double> c;
87 // rhs vectors
88 std::vector<double> dx, dy;
89 // solution vectors
90 std::vector<double> fx(waypoints.size() - 2, 0.0),
91 fy(waypoints.size() - 2, 0.0);
92
93 // populate above-diagonal and below-diagonal vectors
94 a.emplace_back(0);
95 for (size_t i = 0; i < waypoints.size() - 3; ++i) {
96 a.emplace_back(1);
97 c.emplace_back(1);
98 }
99 c.emplace_back(0);
100
101 // populate rhs vectors
102 dx.emplace_back(3 * (waypoints[2].x() - waypoints[0].x()) - x_initial[1]);
103 dy.emplace_back(3 * (waypoints[2].y() - waypoints[0].y()) - y_initial[1]);
104 if (waypoints.size() > 4) {
105 for (size_t i = 1; i <= waypoints.size() - 4; ++i) {
106 // dx and dy represent the derivatives of the internal waypoints. The
107 // derivative of the second internal waypoint should involve the third
108 // and first internal waypoint, which have indices of 1 and 3 in the
109 // waypoints list (which contains ALL waypoints).
110 dx.emplace_back(3 * (waypoints[i + 2].x() - waypoints[i].x()));
111 dy.emplace_back(3 * (waypoints[i + 2].y() - waypoints[i].y()));
112 }
113 }
114 dx.emplace_back(3 * (waypoints[waypoints.size() - 1].x() -
115 waypoints[waypoints.size() - 3].x()) -
116 x_final[1]);
117 dy.emplace_back(3 * (waypoints[waypoints.size() - 1].y() -
118 waypoints[waypoints.size() - 3].y()) -
119 y_final[1]);
120
121 // Compute solution to tridiagonal system
122 thomas_algorithm(a, b, c, dx, &fx);
123 thomas_algorithm(a, b, c, dy, &fy);
124
125 fx.emplace(fx.begin(), x_initial[1]);
126 fx.emplace_back(x_final[1]);
127 fy.emplace(fy.begin(), y_initial[1]);
128 fy.emplace_back(y_final[1]);
129
130 for (size_t i = 0; i < fx.size() - 1; ++i) {
131 // Create the spline.
132 const CubicHermiteSpline spline{{waypoints[i].x(), fx[i]},
133 {waypoints[i + 1].x(), fx[i + 1]},
134 {waypoints[i].y(), fy[i]},
135 {waypoints[i + 1].y(), fy[i + 1]}};
136
137 splines.push_back(spline);
138 }
139 } else if (waypoints.size() == 1) {
140 const double x_deriv =
141 (3 * (x_final[0] - x_initial[0]) - x_final[1] - x_initial[1]) / 4.0;
142 const double y_deriv =
143 (3 * (y_final[0] - y_initial[0]) - y_final[1] - y_initial[1]) / 4.0;
144
145 std::array<double, 2> mid_x_control_vector{waypoints[0].x(), x_deriv};
146 std::array<double, 2> mid_y_control_vector{waypoints[0].y(), y_deriv};
147
148 splines.emplace_back(x_initial, mid_x_control_vector, y_initial,
149 mid_y_control_vector);
150 splines.emplace_back(mid_x_control_vector, x_final, mid_y_control_vector,
151 y_final);
152
153 } else {
154 // Create the spline.
155 const CubicHermiteSpline spline{x_initial, x_final, y_initial, y_final};
156 splines.push_back(spline);
157 }
158
159 return splines;
160 }

162 private:
163 static Spline<3>::ControlVector cubic_control_vector(double scalar,
164 const Pose2d& point) {
165 return {{point.x(), scalar * point.rotation().cos()},
166 {point.y(), scalar * point.rotation().sin()}};
167 }

176 static void thomas_algorithm(const std::vector<double>& a,
177 const std::vector<double>& b,
178 const std::vector<double>& c,
179 const std::vector<double>& d,
180 std::vector<double>* solution_vector) {
181 auto& f = *solution_vector;
182 size_t N = d.size();
183
184 // Create the temporary vectors
185 // Note that this is inefficient as it is possible to call
186 // this function many times. A better implementation would
187 // pass these temporary matrices by non-const reference to
188 // save excess allocation and deallocation
189 std::vector<double> c_star(N, 0.0);
190 std::vector<double> d_star(N, 0.0);
191
192 // This updates the coefficients in the first row
193 // Note that we should be checking for division by zero here
194 c_star[0] = c[0] / b[0];
195 d_star[0] = d[0] / b[0];
196
197 // Create the c_star and d_star coefficients in the forward sweep
198 for (size_t i = 1; i < N; ++i) {
199 double m = 1.0 / (b[i] - a[i] * c_star[i - 1]);
200 c_star[i] = c[i] * m;
201 d_star[i] = (d[i] - a[i] * d_star[i - 1]) * m;
202 }
203
204 f[N - 1] = d_star[N - 1];
205 // This is the reverse sweep, used to update the solution vector f
206 for (int i = N - 2; i >= 0; i--) {
207 f[i] = d_star[i] - c_star[i] * f[i + 1];
208 }
209 }
210};
