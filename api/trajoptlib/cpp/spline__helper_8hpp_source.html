32 cubic_control_vectors_from_waypoints(
33 const Pose2d& start, const std::vector<Translation2d>& interior_waypoints,
34 const Pose2d& end) {
35 double scalar;
36 if (interior_waypoints.empty()) {
37 scalar = 1.2 * start.translation().distance(end.translation());
38 } else {
39 scalar = 1.2 * start.translation().distance(interior_waypoints.front());
40 }
41 const auto initial_control_vector = cubic_control_vector(scalar, start);
42 if (!interior_waypoints.empty()) {

46 return {initial_control_vector, final_control_vector};

66 static std::vector<CubicHermiteSpline> cubic_splines_from_control_vectors(
67 const Spline<3>::ControlVector& start,
68 std::vector<Translation2d> waypoints,
69 const Spline<3>::ControlVector& end) {
70 std::vector<CubicHermiteSpline> splines;
71
72 std::array<double, 2> x_initial = start.x;
73 std::array<double, 2> y_initial = start.y;
74 std::array<double, 2> x_final = end.x;
75 std::array<double, 2> y_final = end.y;
76
77 if (waypoints.size() > 1) {
78 waypoints.emplace(waypoints.begin(),
79 Translation2d{x_initial[0], y_initial[0]});
80 waypoints.emplace_back(Translation2d{x_final[0], y_final[0]});
81
82 // Populate tridiagonal system for clamped cubic
83 /* See:
84 https://www.uio.no/studier/emner/matnat/ifi/nedlagte-emner/INF-MAT4350/h08
85 /undervisningsmateriale/chap7alecture.pdf
86 */
87
88 // Above-diagonal of tridiagonal matrix, zero-padded
89 std::vector<double> a;
90 // Diagonal of tridiagonal matrix
91 std::vector<double> b(waypoints.size() - 2, 4.0);
92 // Below-diagonal of tridiagonal matrix, zero-padded
93 std::vector<double> c;
94 // rhs vectors
95 std::vector<double> dx, dy;
96 // solution vectors
97 std::vector<double> fx(waypoints.size() - 2, 0.0),
98 fy(waypoints.size() - 2, 0.0);
99
100 // populate above-diagonal and below-diagonal vectors
101 a.emplace_back(0);
102 for (size_t i = 0; i < waypoints.size() - 3; ++i) {
103 a.emplace_back(1);
104 c.emplace_back(1);
105 }
106 c.emplace_back(0);
107
108 // populate rhs vectors
109 dx.emplace_back(3 * (waypoints[2].x() - waypoints[0].x()) - x_initial[1]);
110 dy.emplace_back(3 * (waypoints[2].y() - waypoints[0].y()) - y_initial[1]);
111 if (waypoints.size() > 4) {
112 for (size_t i = 1; i <= waypoints.size() - 4; ++i) {
113 // dx and dy represent the derivatives of the internal waypoints. The
114 // derivative of the second internal waypoint should involve the third
115 // and first internal waypoint, which have indices of 1 and 3 in the
116 // waypoints list (which contains ALL waypoints).
117 dx.emplace_back(3 * (waypoints[i + 2].x() - waypoints[i].x()));
118 dy.emplace_back(3 * (waypoints[i + 2].y() - waypoints[i].y()));
119 }
120 }
121 dx.emplace_back(3 * (waypoints[waypoints.size() - 1].x() -
122 waypoints[waypoints.size() - 3].x()) -
123 x_final[1]);
124 dy.emplace_back(3 * (waypoints[waypoints.size() - 1].y() -
125 waypoints[waypoints.size() - 3].y()) -
126 y_final[1]);
127
128 // Compute solution to tridiagonal system
129 thomas_algorithm(a, b, c, dx, &fx);
130 thomas_algorithm(a, b, c, dy, &fy);
131
132 fx.emplace(fx.begin(), x_initial[1]);
133 fx.emplace_back(x_final[1]);
134 fy.emplace(fy.begin(), y_initial[1]);
135 fy.emplace_back(y_final[1]);
136
137 for (size_t i = 0; i < fx.size() - 1; ++i) {
138 // Create the spline.
139 const CubicHermiteSpline spline{{waypoints[i].x(), fx[i]},
140 {waypoints[i + 1].x(), fx[i + 1]},
141 {waypoints[i].y(), fy[i]},
142 {waypoints[i + 1].y(), fy[i + 1]}};
143
144 splines.push_back(spline);
145 }
146 } else if (waypoints.size() == 1) {
147 const double x_deriv =
148 (3 * (x_final[0] - x_initial[0]) - x_final[1] - x_initial[1]) / 4.0;
149 const double y_deriv =
150 (3 * (y_final[0] - y_initial[0]) - y_final[1] - y_initial[1]) / 4.0;
151
152 std::array<double, 2> mid_x_control_vector{waypoints[0].x(), x_deriv};
153 std::array<double, 2> mid_y_control_vector{waypoints[0].y(), y_deriv};
154
155 splines.emplace_back(x_initial, mid_x_control_vector, y_initial,
156 mid_y_control_vector);
157 splines.emplace_back(mid_x_control_vector, x_final, mid_y_control_vector,
158 y_final);
159
160 } else {
161 // Create the spline.
162 const CubicHermiteSpline spline{x_initial, x_final, y_initial, y_final};
163 splines.push_back(spline);
164 }
165
166 return splines;
167 }

170 static Spline<3>::ControlVector cubic_control_vector(double scalar,
171 const Pose2d& point) {
172 return {{point.x(), scalar * point.rotation().cos()},
173 {point.y(), scalar * point.rotation().sin()}};
174 }

185 static void thomas_algorithm(const std::vector<double>& a,
186 const std::vector<double>& b,
187 const std::vector<double>& c,
188 const std::vector<double>& d,
189 std::vector<double>* solution_vector) {
190 auto& f = *solution_vector;
191 size_t N = d.size();
192
193 // Create the temporary vectors
194 // Note that this is inefficient as it is possible to call
195 // this function many times. A better implementation would
196 // pass these temporary matrices by non-const reference to
197 // save excess allocation and deallocation
198 std::vector<double> c_star(N, 0.0);
199 std::vector<double> d_star(N, 0.0);
200
201 // This updates the coefficients in the first row
202 // Note that we should be checking for division by zero here
203 c_star[0] = c[0] / b[0];
204 d_star[0] = d[0] / b[0];
205
206 // Create the c_star and d_star coefficients in the forward sweep
207 for (size_t i = 1; i < N; ++i) {
208 double m = 1.0 / (b[i] - a[i] * c_star[i - 1]);
209 c_star[i] = c[i] * m;
210 d_star[i] = (d[i] - a[i] * d_star[i - 1]) * m;
211 }
212
213 f[N - 1] = d_star[N - 1];
214 // This is the reverse sweep, used to update the solution vector f
215 for (int i = N - 2; i >= 0; i--) {
216 f[i] = d_star[i] - c_star[i] * f[i + 1];
217 }
218 }
219};
