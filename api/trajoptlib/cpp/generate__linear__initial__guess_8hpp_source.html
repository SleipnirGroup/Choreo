23template <typename Solution>
24inline Solution generate_linear_initial_guess(
25 const std::vector<std::vector<Pose2d>>& initial_guess_points,
26 const std::vector<size_t> control_interval_counts) {
27 size_t wpt_cnt = control_interval_counts.size() + 1;
28 size_t samp_tot = get_index(control_interval_counts, wpt_cnt - 1, 0) + 1;
29
30 Solution initial_guess;
31
32 initial_guess.x.reserve(samp_tot);
33 initial_guess.y.reserve(samp_tot);
34 if constexpr (std::same_as<Solution, DifferentialSolution>) {
35 initial_guess.heading.reserve(samp_tot);
36 } else {
37 initial_guess.thetacos.reserve(samp_tot);
38 initial_guess.thetasin.reserve(samp_tot);
39 }
40
41 initial_guess.dt.reserve(samp_tot);
42
43 initial_guess.x.push_back(initial_guess_points.front().front().x());
44 initial_guess.y.push_back(initial_guess_points.front().front().y());
45 if constexpr (std::same_as<Solution, DifferentialSolution>) {
46 initial_guess.heading.push_back(
47 initial_guess_points.front().front().rotation().radians());
48 } else {
49 initial_guess.thetacos.push_back(
50 initial_guess_points.front().front().rotation().cos());
51 initial_guess.thetasin.push_back(
52 initial_guess_points.front().front().rotation().sin());
53 }
54
55 for (size_t i = 0; i < samp_tot; ++i) {
56 initial_guess.dt.push_back((wpt_cnt * 5.0) / samp_tot);
57 }
58
59 for (size_t wpt_index = 1; wpt_index < wpt_cnt; ++wpt_index) {
60 size_t N_sgmt = control_interval_counts.at(wpt_index - 1);
61 size_t guess_point_count = initial_guess_points.at(wpt_index).size();
62 size_t N_guess_sgmt = N_sgmt / guess_point_count;
63 append_range(
64 initial_guess.x,
65 linspace(initial_guess_points.at(wpt_index - 1).back().x(),
66 initial_guess_points.at(wpt_index).front().x(), N_guess_sgmt));
67 append_range(
68 initial_guess.y,
69 linspace(initial_guess_points.at(wpt_index - 1).back().y(),
70 initial_guess_points.at(wpt_index).front().y(), N_guess_sgmt));
71 auto wpt_thetas = angle_linspace(
72 initial_guess_points.at(wpt_index - 1).back().rotation().radians(),
73 initial_guess_points.at(wpt_index).front().rotation().radians(),
74 N_guess_sgmt);
75 for (auto theta : wpt_thetas) {
76 if constexpr (std::same_as<Solution, DifferentialSolution>) {
77 initial_guess.heading.push_back(theta);
78 } else {
79 initial_guess.thetacos.push_back(std::cos(theta));
80 initial_guess.thetasin.push_back(std::sin(theta));
81 }
82 }
83 for (size_t guess_point_index = 1;
84 guess_point_index < guess_point_count - 1;
85 ++guess_point_index) { // if three or more guess points
86 append_range(
87 initial_guess.x,
88 linspace(
89 initial_guess_points.at(wpt_index).at(guess_point_index - 1).x(),
90 initial_guess_points.at(wpt_index).at(guess_point_index).x(),
91 N_guess_sgmt));
92 append_range(
93 initial_guess.y,
94 linspace(
95 initial_guess_points.at(wpt_index).at(guess_point_index - 1).y(),
96 initial_guess_points.at(wpt_index).at(guess_point_index).y(),
97 N_guess_sgmt));
98 auto guess_thetas = angle_linspace(initial_guess_points.at(wpt_index)
99 .at(guess_point_index - 1)
100 .rotation()
101 .radians(),
102 initial_guess_points.at(wpt_index)
103 .at(guess_point_index)
104 .rotation()
105 .radians(),
106 N_guess_sgmt);
107 for (auto theta : guess_thetas) {
108 if constexpr (std::same_as<Solution, DifferentialSolution>) {
109 initial_guess.heading.push_back(theta);
110 } else {
111 initial_guess.thetacos.push_back(std::cos(theta));
112 initial_guess.thetasin.push_back(std::sin(theta));
113 }
114 }
115 }
116 if (guess_point_count > 1) { // if two or more guess points
117 size_t N_last_guess_sgmt =
118 N_sgmt - (guess_point_count - 1) * N_guess_sgmt;
119 append_range(
120 initial_guess.x,
121 linspace(
122 initial_guess_points.at(wpt_index).at(guess_point_count - 2).x(),
123 initial_guess_points.at(wpt_index).back().x(),
124 N_last_guess_sgmt));
125 append_range(
126 initial_guess.y,
127 linspace(
128 initial_guess_points.at(wpt_index).at(guess_point_count - 2).y(),
129 initial_guess_points.at(wpt_index).back().y(),
130 N_last_guess_sgmt));
131 auto last_thetas = angle_linspace(
132 initial_guess_points.at(wpt_index)
133 .at(guess_point_count - 2)
134 .rotation()
135 .radians(),
136 initial_guess_points.at(wpt_index).back().rotation().radians(),
137 N_last_guess_sgmt);
138 for (auto theta : last_thetas) {
139 if constexpr (std::same_as<Solution, DifferentialSolution>) {
140 initial_guess.heading.push_back(theta);
141 } else {
142 initial_guess.thetacos.push_back(std::cos(theta));
143 initial_guess.thetasin.push_back(std::sin(theta));
144 }
145 }
146 }
147 }
148
149 return initial_guess;
150}
