24template <typename Solution>
25inline std::vector<CubicHermitePoseSplineHolonomic> splines_from_waypoints(
26 const std::vector<std::vector<Pose2d>> initial_guess_points) {
27 size_t total_guess_points = 0;
28 for (const auto& points : initial_guess_points) {
29 total_guess_points += points.size();
30 }
31 std::vector<Translation2d> flat_translation_points;
32 std::vector<Rotation2d> flat_headings;
33 flat_translation_points.reserve(total_guess_points);
34 flat_headings.reserve(total_guess_points);
35
36 // populate translation and heading vectors
37 for (const auto& guess_points : initial_guess_points) {
38 for (const auto& guess_point : guess_points) {
39 flat_translation_points.emplace_back(guess_point.translation().x(),
40 guess_point.translation().y());
41 flat_headings.emplace_back(guess_point.rotation().cos(),
42 guess_point.rotation().sin());
43 }
44 }
45
46 std::vector<CubicHermiteSpline> splines_temp;
47 splines_temp.reserve(total_guess_points);
48
49 if constexpr (std::same_as<Solution, DifferentialSolution>) {
50 for (size_t i = 1; i < flat_translation_points.size(); ++i) {
51 const auto spline_control_vectors =
52 SplineHelper::cubic_control_vectors_from_waypoints(
53 Pose2d{flat_translation_points.at(i - 1),
54 flat_headings.at(i - 1)},
55 {}, Pose2d{flat_translation_points.at(i), flat_headings.at(i)});
56 const auto s = SplineHelper::cubic_splines_from_control_vectors(
57 spline_control_vectors.front(), {}, spline_control_vectors.back());
58 for (const auto& _s : s) {
59 splines_temp.push_back(_s);
60 }
61 }
62 } else {
63 // calculate angles and pose for start and end of path spline
64 const auto start_spline_angle =
65 (flat_translation_points.at(1) - flat_translation_points.at(0)).angle();
66 const auto end_spline_angle =
67 (flat_translation_points.back() -
68 flat_translation_points.at(flat_translation_points.size() - 2))
69 .angle();
70 const Pose2d start{flat_translation_points.front(), start_spline_angle};
71 const Pose2d end{flat_translation_points.back(), end_spline_angle};
72
73 // use all interior points to create the path spline
74 std::vector<Translation2d> interior_points{
75 flat_translation_points.begin() + 1, flat_translation_points.end() - 1};
76
77 const auto spline_control_vectors =
78 SplineHelper::cubic_control_vectors_from_waypoints(
79 start, interior_points, end);
80 const auto s = SplineHelper::cubic_splines_from_control_vectors(
81 spline_control_vectors.front(), interior_points,
82 spline_control_vectors.back());
83 for (const auto& _s : s) {
84 splines_temp.push_back(_s);
85 }
86 }
87
88 std::vector<CubicHermitePoseSplineHolonomic> splines;
89 splines.reserve(splines_temp.size());
90 for (size_t i = 1; i <= splines_temp.size(); ++i) {
91 splines.emplace_back(splines_temp.at(i - 1).get_initial_control_vector().x,
92 splines_temp.at(i - 1).get_final_control_vector().x,
93 splines_temp.at(i - 1).get_initial_control_vector().y,
94 splines_temp.at(i - 1).get_final_control_vector().y,
95 flat_headings.at(i - 1), flat_headings.at(i));
96 }
97 return splines;
98}

100template <typename Solution>
101inline Solution generate_spline_initial_guess(
102 const std::vector<std::vector<Pose2d>>& initial_guess_points,
103 const std::vector<size_t> control_interval_counts) {
104 std::vector<CubicHermitePoseSplineHolonomic> splines =
105 splines_from_waypoints<Solution>(initial_guess_points);
106 std::vector<std::vector<PoseWithCurvature>> sgmt_points;
107 for (size_t i = 0; i < initial_guess_points.size(); ++i) {
108 for ([[maybe_unused]]
109 size_t j = 0;
110 j < initial_guess_points.at(i).size(); ++j) {
111 sgmt_points.push_back(std::vector<PoseWithCurvature>());
112 }
113 }
114
115 size_t traj_idx = 0;
116 if constexpr (std::same_as<Solution, DifferentialSolution>) {
117 sgmt_points.at(0).push_back(
118 splines.at(traj_idx).get_point(0, true).value());
119 } else {
120 sgmt_points.at(0).push_back(
121 splines.at(traj_idx).get_point(0, false).value());
122 }
123 for (size_t sgmt_idx = 1; sgmt_idx < initial_guess_points.size();
124 ++sgmt_idx) {
125 auto guess_points_size = initial_guess_points.at(sgmt_idx).size();
126 auto samples_for_sgmt = control_interval_counts.at(sgmt_idx - 1);
127 size_t samples = samples_for_sgmt / guess_points_size;
128 for (size_t guessIdx = 0; guessIdx < guess_points_size; ++guessIdx) {
129 if (guessIdx == (guess_points_size - 1)) {
130 samples += (samples_for_sgmt % guess_points_size);
131 }
132 for (size_t sample_idx = 1; sample_idx < samples + 1; ++sample_idx) {
133 auto t = static_cast<double>(sample_idx) / samples;
134
135 if constexpr (std::same_as<Solution, DifferentialSolution>) {
136 const auto state = splines.at(traj_idx).get_point(t, true).value();
137 sgmt_points.at(traj_idx + 1).push_back(state);
138 } else {
139 const auto state = splines.at(traj_idx).get_point(t, false).value();
140 sgmt_points.at(traj_idx + 1).push_back(state);
141 }
142 }
143 ++traj_idx;
144 }
145 }
146
147 size_t wpt_cnt = control_interval_counts.size() + 1;
148 size_t samp_tot = get_index(control_interval_counts, wpt_cnt - 1, 0) + 1;
149
150 Solution initial_guess;
151
152 initial_guess.x.reserve(samp_tot);
153 initial_guess.y.reserve(samp_tot);
154 if constexpr (std::same_as<Solution, DifferentialSolution>) {
155 initial_guess.heading.reserve(samp_tot);
156 } else {
157 initial_guess.thetacos.reserve(samp_tot);
158 initial_guess.thetasin.reserve(samp_tot);
159 }
160
161 initial_guess.dt.reserve(samp_tot);
162 for (size_t i = 0; i < samp_tot; ++i) {
163 initial_guess.dt.push_back((wpt_cnt * 5.0) / samp_tot);
164 }
165
166 for (auto sgmt : sgmt_points) {
167 for (auto pt : sgmt) {
168 initial_guess.x.push_back(pt.first.x());
169 initial_guess.y.push_back(pt.first.y());
170 if constexpr (std::same_as<Solution, DifferentialSolution>) {
171 initial_guess.heading.push_back(pt.first.rotation().radians());
172 } else {
173 initial_guess.thetacos.push_back(pt.first.rotation().cos());
174 initial_guess.thetasin.push_back(pt.first.rotation().sin());
175 }
176 }
177 }
178 return initial_guess;
179}
