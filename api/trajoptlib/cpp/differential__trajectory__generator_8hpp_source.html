170class TRAJOPT_DLLEXPORT DifferentialTrajectory {
171 public:
173 std::vector<DifferentialTrajectorySample> samples;
174
175 DifferentialTrajectory() = default;
176
182 explicit DifferentialTrajectory(
183 std::vector<DifferentialTrajectorySample> samples)
184 : samples{std::move(samples)} {}
185
191 explicit DifferentialTrajectory(const DifferentialSolution& solution) {
192 double ts = 0.0;
193 for (size_t sample = 0; sample < solution.x.size(); ++sample) {
194 samples.emplace_back(
195 ts, solution.x[sample], solution.y[sample], solution.heading[sample],
196 solution.vl[sample], solution.vr[sample],
197 solution.angular_velocity[sample], solution.al[sample],
198 solution.ar[sample], solution.angular_acceleration[sample],
199 solution.Fl[sample], solution.Fr[sample]);
200 ts += solution.dt[sample];
201 }
202 }
203};
