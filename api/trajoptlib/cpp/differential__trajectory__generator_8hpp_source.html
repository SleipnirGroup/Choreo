160class TRAJOPT_DLLEXPORT DifferentialTrajectory {
161 public:
163 std::vector<DifferentialTrajectorySample> samples;
164
165 DifferentialTrajectory() = default;
166
170 explicit DifferentialTrajectory(
171 std::vector<DifferentialTrajectorySample> samples)
172 : samples{std::move(samples)} {}
173
177 explicit DifferentialTrajectory(const DifferentialSolution& solution) {
178 double ts = 0.0;
179 for (size_t sample = 0; sample < solution.x.size(); ++sample) {
180 samples.emplace_back(
181 ts, solution.x[sample], solution.y[sample], solution.heading[sample],
182 solution.vl[sample], solution.vr[sample],
183 solution.angular_velocity[sample], solution.al[sample],
184 solution.ar[sample], solution.angular_acceleration[sample],
185 solution.Fl[sample], solution.Fr[sample]);
186 ts += solution.dt[sample];
187 }
188 }
189};
