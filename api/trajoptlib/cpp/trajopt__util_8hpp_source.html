34inline std::vector<double> linspace(double start, double end,
35 size_t num_samples) {
36 std::vector<double> result;
37 double delta = (end - start) / num_samples;
38 for (size_t i = 1; i <= num_samples; ++i) {
39 result.push_back(start + i * delta);
40 }
41 return result;
42}

49constexpr double input_modulus(double input, double minimum_input,
50 double maximum_input) {
51 double modulus = maximum_input - minimum_input;
52
53 // Wrap input if it's above the maximum input
54 int numMax = (input - minimum_input) / modulus;
55 input -= numMax * modulus;
56
57 // Wrap input if it's below the minimum input
58 int numMin = (input - maximum_input) / modulus;
59 input -= numMin * modulus;
60
61 return input;
62}

67constexpr double angle_modulus(double angle) {
68 return input_modulus(angle, -std::numbers::pi, std::numbers::pi);
69}

79inline std::vector<double> angle_linspace(double start, double end,
80 size_t num_samples) {
81 return linspace(start, start + angle_modulus(end - start), num_samples);
82}

90inline double calculate_trapezoidal_time(double distance, double velocity,
91 double acceleration) {
92 if (distance > ((velocity * velocity) / acceleration)) {
93 // Velocity profile is shaped like a trapezoid
94 return distance / velocity + velocity / acceleration;
95 } else {
96 // Velocity profile is shaped like a triangle
97 return 2.0 * std::sqrt(distance * acceleration) / acceleration;
98 }
99}
