38inline std::vector<double> linspace(double start, double end,
39 size_t num_samples) {
40 std::vector<double> result;
41 double delta = (end - start) / num_samples;
42 for (size_t i = 1; i <= num_samples; ++i) {
43 result.push_back(start + i * delta);
44 }
45 return result;
46}

55constexpr double input_modulus(double input, double minimum_input,
56 double maximum_input) {
57 double modulus = maximum_input - minimum_input;
58
59 // Wrap input if it's above the maximum input
60 int numMax = (input - minimum_input) / modulus;
61 input -= numMax * modulus;
62
63 // Wrap input if it's below the minimum input
64 int numMin = (input - maximum_input) / modulus;
65 input -= numMin * modulus;
66
67 return input;
68}

75constexpr double angle_modulus(double angle) {
76 return input_modulus(angle, -std::numbers::pi, std::numbers::pi);
77}

89inline std::vector<double> angle_linspace(double start, double end,
90 size_t num_samples) {
91 return linspace(start, start + angle_modulus(end - start), num_samples);
92}

102inline double calculate_trapezoidal_time(double distance, double velocity,
103 double acceleration) {
104 if (distance > ((velocity * velocity) / acceleration)) {
105 // Velocity profile is shaped like a trapezoid
106 return distance / velocity + velocity / acceleration;
107 } else {
108 // Velocity profile is shaped like a triangle
109 return 2.0 * std::sqrt(distance * acceleration) / acceleration;
110 }
111}
