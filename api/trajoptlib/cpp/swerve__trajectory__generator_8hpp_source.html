163class TRAJOPT_DLLEXPORT SwerveTrajectory {
164 public:
166 std::vector<SwerveTrajectorySample> samples;
167
168 SwerveTrajectory() = default;
169
173 explicit SwerveTrajectory(std::vector<SwerveTrajectorySample> samples)
174 : samples{std::move(samples)} {}
175
179 explicit SwerveTrajectory(const SwerveSolution& solution) {
180 double ts = 0.0;
181 for (size_t sample = 0; sample < solution.x.size(); ++sample) {
182 samples.emplace_back(
183 ts, solution.x[sample], solution.y[sample],
184 std::atan2(solution.thetasin[sample], solution.thetacos[sample]),
185 solution.vx[sample], solution.vy[sample], solution.omega[sample],
186 solution.ax[sample], solution.ay[sample], solution.alpha[sample],
187 solution.module_fx[sample], solution.module_fy[sample]);
188 ts += solution.dt[sample];
189 }
190 }
191};
